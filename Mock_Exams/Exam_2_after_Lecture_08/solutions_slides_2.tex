\begin{frame}{Problem 1 --- restatement}
  \begin{exercise}[Problem 1 --- Logistic regression by hand (18 P)]
    A bank models a credit-card customer's default probability from the monthly balance $x$
    (EUR, range 0--400) by logistic regression, fitted in Python with \texttt{statsmodels}
    (\texttt{sm.Logit(y, X).fit()}):
    \begin{center}\scriptsize
    \begin{tabular}{lrrrr}
      \toprule
       & coef & std err & $z$ & $P>|z|$\\
      \midrule
      \texttt{const}   & $-4.0000$ & 0.400 & $-10.000$ & $<0.0001$\\
      \texttt{balance} & $\phantom{-}0.0200$ & 0.002 & $\phantom{-}10.000$ & $<0.0001$\\
      \bottomrule
    \end{tabular}
    \end{center}
    \begin{itemize}
      \item[(a)] Read $\hat{\beta}_0,\hat{\beta}_1$ from the output; predicted default
        probability at $x=100$ and $x=250$. \hfill (4 P)
      \item[(b)] Odds and log-odds at both balances ($x=100$, $x=250$); interpret the odds
        at $x=250$. \hfill (4 P)
      \item[(c)] Balance $x^{*}$ with $\hat{p}(x^{*})=0.5$. \hfill (3 P)
      \item[(d)] Odds factor per EUR and per 100 EUR; why is ``probability $+0.02$ per EUR''
        wrong? \hfill (4 P)
      \item[(e)] Two reasons why linear regression on the 0/1 response is inappropriate. \hfill (3 P)
    \end{itemize}
  \end{exercise}
\end{frame}

\begin{frame}{Problem 1 --- solution (a): predicted probabilities}
  \begin{solutionbox}[Solution 1(a) --- read the coefficients then substitute]
    \textbf{Step 1 --- read the fitted model.} From the \texttt{statsmodels} output,
    $\hat{\beta}_0=-4$ (row \texttt{const}) and $\hat{\beta}_1=0.02$ (row \texttt{balance}).
    The fitted linear predictor (the log-odds) and the probability are
    \[
      \hat{\eta}(x)=\hat{\beta}_0+\hat{\beta}_1 x=-4+0.02\,x,
      \qquad
      \hat{p}(x)=\frac{e^{\hat{\eta}(x)}}{1+e^{\hat{\eta}(x)}} .
    \]
    \textbf{Step 2 --- evaluate the linear predictor.}\quad
    $\hat{\eta}(100)=-4+0.02\cdot 100=-2$\ \ and\ \ $\hat{\eta}(250)=-4+0.02\cdot 250=1$.\\[0.5mm]
    \textbf{Step 3 --- push through the logistic (sigmoid) function.}
    \[
      \hat{p}(100)=\frac{e^{-2}}{1+e^{-2}}=\frac{0.1353}{1.1353}=\mathbf{0.119},
      \qquad
      \hat{p}(250)=\frac{e^{1}}{1+e^{1}}=\frac{2.7183}{3.7183}=\mathbf{0.731}.
    \]
    So the estimated default risk is $11.9\,\%$ at $100$ EUR and $73.1\,\%$ at $250$ EUR; the
    sigmoid keeps every probability inside $(0,1)$.
  \end{solutionbox}
\end{frame}

\begin{frame}{Problem 1 --- solution (b): odds and log-odds}
  \begin{solutionbox}[Solution 1(b) --- odds and log-odds at both balances]
    The \emph{log-odds} equal the linear predictor and the \emph{odds} are its exponential:
    \[
      \log\!\frac{\hat{p}(x)}{1-\hat{p}(x)}=\hat{\eta}(x),
      \qquad
      \frac{\hat{p}(x)}{1-\hat{p}(x)}=e^{\hat{\eta}(x)} .
    \]
    \begin{center}
    \begin{tabular}{lcc}
      \toprule
       & $x=100$ & $x=250$\\
      \midrule
      log-odds $\hat{\eta}(x)$ & $-2.000$ & $1.000$\\
      odds $e^{\hat{\eta}(x)}$ & $e^{-2}=0.135$ & $e^{1}=2.718$\\
      \bottomrule
    \end{tabular}
    \end{center}
    \textbf{Cross-check with (a):}\quad
    $\dfrac{0.119}{0.881}=0.135$~\checkmark\qquad
    $\dfrac{0.731}{0.269}=2.718$~\checkmark\\[0.5mm]
    \emph{Interpretation ($x=250$):} the odds are about $2.72$ to $1$ --- default is roughly
    $2.7$ times as probable as non-default for this customer.
  \end{solutionbox}
\end{frame}

\begin{frame}{Problem 1 --- solution (c): the 50:50 balance}
  \begin{solutionbox}[Solution 1(c) --- solve for the 50:50 balance]
    A predicted probability of exactly one half means even odds, hence zero log-odds:
    \[
      \hat{p}(x^{*})=0.5
      \iff \frac{\hat{p}}{1-\hat{p}}=1
      \iff \hat{\eta}(x^{*})=\log 1=0 .
    \]
    Set the linear predictor to zero and solve for the balance:
    \[
      -4+0.02\,x^{*}=0
      \;\Longrightarrow\;
      0.02\,x^{*}=4
      \;\Longrightarrow\;
      x^{*}=\frac{4}{0.02}=\mathbf{200\ \text{EUR}} .
    \]
    In general $x^{*}=-\hat{\beta}_0/\hat{\beta}_1$.
  \end{solutionbox}
  \begin{takeaway}[Decision boundary]
    At the default threshold $0.5$: below $200$ EUR the model predicts ``no default'' as
    more likely, above $200$ EUR ``default''. So $x^{*}=200$ is the classifier's decision
    boundary.
  \end{takeaway}
\end{frame}

\begin{frame}{Problem 1 --- solution (d): the slope on the odds scale}
  \begin{solutionbox}[Solution 1(d) --- the slope on the odds scale]
    Raising the balance by $\Delta$ EUR adds $\hat{\beta}_1\Delta$ to the log-odds and hence
    \emph{multiplies} the odds by $e^{\hat{\beta}_1\Delta}$:
    \[
      \text{per }+1\text{ EUR: } e^{\hat{\beta}_1}=e^{0.02}=\mathbf{1.020}\ (+2.0\,\%),
      \qquad
      \text{per }+100\text{ EUR: } e^{100\hat{\beta}_1}=e^{2}=\mathbf{7.389}\ (+639\,\%).
    \]
    \textbf{Why ``probability $+0.02$ per EUR'' is wrong.} $\hat{\beta}_1$ is the change in
    the \emph{log-odds}, not in the probability. The probability slope
    \[
      \frac{\partial \hat{p}}{\partial x}=\hat{\beta}_1\,\hat{p}(x)\bigl(1-\hat{p}(x)\bigr)
    \]
    \emph{depends on $x$}: it is largest at $\hat{p}=0.5$ (here $x=200$:
    $0.02\cdot 0.25=0.005$ per EUR) and nearly $0$ in the tails --- there is no constant
    ``probability effect per EUR''.
  \end{solutionbox}
\end{frame}

\begin{frame}{Problem 1 --- solution (e): why not linear regression}
  \begin{solutionbox}[Solution 1(e) --- two reasons the linear probability model fails]
    \begin{itemize}
      \item[(1)] \textbf{Out-of-range predictions.} The straight line
        $\hat{\beta}_0+\hat{\beta}_1x$ is unbounded, so for small or large balances it
        returns ``probabilities'' below $0$ or above $1$ --- not sensible estimates of
        $\Pr(Y=1\mid x)$. The logistic function is guaranteed to stay in $[0,1]$.
      \item[(2)] \textbf{Violated error assumptions.} A $0/1$ response cannot have normal,
        homoscedastic errors: its conditional variance
        $\operatorname{Var}(Y\mid x)=p(x)\bigl(1-p(x)\bigr)$ depends on $x$
        (heteroscedasticity), so the usual OLS standard errors and inference are unreliable.
      \item[] \emph{(Bonus:} with $>2$ classes a numeric coding of $Y$ also imposes an
        artificial ordering and equal spacing between classes.\emph{)}
    \end{itemize}
  \end{solutionbox}
  \begin{takeaway}[Remember]
    Logistic regression is a \emph{linear model for the log-odds}. Coefficients are additive
    on the log-odds scale and multiplicative on the odds scale --- never on the probability
    scale.
  \end{takeaway}
\end{frame}

\begin{frame}{Problem 2 --- restatement}
  \begin{exercise}[Problem 2 --- Discriminant analysis and naive Bayes (12 P)]
    \begin{itemize}
      \item[(a)] LDA vs.\ QDA: the distinguishing covariance assumption; number of
        covariance matrices estimated with $K$ classes. \hfill (4 P)
      \item[(b)] Bias--variance: when does LDA beat QDA on test data and vice versa?
        Role of $n$ and of the true class covariances. \hfill (3 P)
      \item[(c)] Key assumption of naive Bayes; why can it help when $p$ is large relative
        to $n$? \hfill (3 P)
      \item[(d)] Which classifiers give linear and which give quadratic or more general
        boundaries? \hfill (2 P)
    \end{itemize}
  \end{exercise}
\end{frame}

\begin{frame}{Problem 2 --- solution (a): LDA vs.\ QDA covariance assumption}
  \begin{solutionbox}[Solution 2(a) --- the distinguishing assumption]
    Both methods model each class density as multivariate normal,
    $X\mid Y=k\sim\mathcal{N}(\mu_k,\Sigma_k)$; they differ \emph{only} in what they assume
    about the covariance matrices:
    \begin{itemize}
      \item \textbf{LDA:} a single, \emph{shared} covariance matrix
        $\Sigma_1=\dots=\Sigma_K=\Sigma$. It estimates \textbf{one} $p\times p$ matrix
        --- $p(p+1)/2$ covariance parameters.
      \item \textbf{QDA:} a \emph{class-specific} covariance matrix $\Sigma_k$ for each
        class. It estimates $\mathbf{K}$ matrices --- about $K\,p(p+1)/2$ covariance
        parameters.
    \end{itemize}
  \end{solutionbox}
  \begin{takeaway}[Parameter count drives everything]
    QDA estimates roughly $K$ times as many covariance parameters as LDA. That extra
    flexibility is exactly what the bias--variance trade-off in part (b) has to weigh.
  \end{takeaway}
\end{frame}

\begin{frame}{Problem 2 --- solution (b): the bias--variance trade-off}
  \begin{solutionbox}[Solution 2(b) --- when does each method win?]
    QDA is the more flexible method (many more parameters): \emph{low bias, high variance}.
    LDA is more restrictive: \emph{low variance, but biased if the $\Sigma_k$ truly differ}.
    \begin{itemize}
      \item \textbf{LDA tends to win} when $n$ is small (or $p$ large), so reducing variance
        is crucial \emph{and} the true class covariances are approximately equal --- then
        LDA's shared-$\Sigma$ assumption costs little bias.
      \item \textbf{QDA tends to win} when $n$ is large, so its many parameters can be
        estimated reliably \emph{and} the class covariances clearly differ --- then LDA's
        assumption would create substantial bias.
    \end{itemize}
  \end{solutionbox}
  \begin{takeaway}[Rule of thumb]
    Small $n$ / similar $\Sigma_k$ $\Rightarrow$ LDA; large $n$ / differing $\Sigma_k$
    $\Rightarrow$ QDA.
  \end{takeaway}
\end{frame}

\begin{frame}{Problem 2 --- solution (c): naive Bayes}
  \begin{solutionbox}[Solution 2(c) --- the conditional-independence assumption]
    \textbf{Key assumption.} Within each class the $p$ predictors are
    \emph{conditionally independent}, so the joint class density factorises into its
    marginals:
    \[
      f_k(x)=f_{k1}(x_1)\,f_{k2}(x_2)\cdots f_{kp}(x_p).
    \]
    \textbf{Why it helps when $p$ is large relative to $n$.} Instead of a full
    $p$-dimensional joint density --- which needs all the within-class associations, e.g.\
    $p(p+1)/2$ covariance parameters per class --- only $p$ \emph{one-dimensional}
    marginals per class are estimated. This drastic cut in the parameter count sharply
    lowers the \emph{variance}. When $p$ is large relative to $n$, that variance saving
    usually outweighs the bias from (falsely) ignoring dependence --- the bias--variance
    trade-off once more.
  \end{solutionbox}
\end{frame}

\begin{frame}{Problem 2 --- solution (d): shape of the decision boundaries}
  \begin{solutionbox}[Solution 2(d) --- shapes of the decision boundaries]
    \begin{itemize}
      \item \textbf{LDA --- linear.} With a shared $\Sigma$ the quadratic term
        $x^{\top}\Sigma^{-1}x$ is common to every class and cancels in the pairwise
        comparison, leaving discriminant functions linear in $x$.
      \item \textbf{QDA --- quadratic.} Each class keeps its own $\Sigma_k$, so the
        quadratic terms do \emph{not} cancel; the boundaries are quadratic curves/surfaces.
      \item \textbf{Naive Bayes --- generally non-linear but additive.} The log-odds take
        the additive form $\sum_j g_j(x_j)$ (no interaction terms). Only in special cases
        (e.g.\ Gaussian marginals with class-shared variances) does it reduce to a linear
        rule.
    \end{itemize}
  \end{solutionbox}
\end{frame}

\begin{frame}{Problem 3 --- restatement}
  \begin{exercise}[Problem 3 --- Confusion matrix and ROC curve (15 P)]
    Logistic-regression default classifier on $n=1{,}000$ customers with threshold
    $\hat{p}>0.5$; labels $1=$ default, $0=$ no default, evaluated with \texttt{scikit-learn}
    (rows $=$ true, columns $=$ predicted, labels in increasing order):
    \begin{center}\footnotesize
    \begin{tabular}{@{}l@{}}
      \toprule
      \ttfamily confusion\_matrix(y\_true, y\_pred)\\
      \ttfamily array([[855,\ \ 45],\\
      \ttfamily \ \ \ \ \ \ \ [\ 40,\ \ 60]])\\
      \bottomrule
    \end{tabular}
    \end{center}
    \begin{itemize}
      \item[(a)] Read TP/FP/FN/TN from the array; accuracy and error rate; sensitivity;
        specificity; precision; FPR; interpret sensitivity and precision. \hfill (7 P)
      \item[(b)] Effect of lowering the threshold from 0.5 to 0.2 on sensitivity and
        specificity; business reason for accepting it. \hfill (4 P)
      \item[(c)] What does the ROC curve plot? Meaning of AUC $=0.5$ and AUC $=1$. \hfill (4 P)
    \end{itemize}
  \end{exercise}
\end{frame}

\begin{frame}{Problem 3 --- solution (a.1): read the confusion matrix}
  \begin{solutionbox}[Solution 3(a) step 1 --- identify the four cells]
    In \texttt{scikit-learn} the \emph{rows} are true classes and the \emph{columns}
    predicted classes, both ordered $0$ (``no default'') then $1$ (``default''). With
    ``default'' $(=1)$ as the positive class:
    \begin{center}
    \renewcommand{\arraystretch}{1.1}
    \begin{tabular}{ll|cc}
      \multicolumn{2}{c}{} & \multicolumn{2}{c}{\textbf{Predicted}}\\
      \multicolumn{2}{c|}{} & $0$ (no def.) & $1$ (default)\\
      \cline{3-4}
      \textbf{True} & $0$ (no def.) & $\text{TN}=855$ & $\text{FP}=45$\\
                    & $1$ (default) & $\text{FN}=40$  & $\text{TP}=60$\\
    \end{tabular}
    \end{center}
    Row sums: $900$ actual non-defaulters and $100$ actual defaulters ($=n=1000$).
  \end{solutionbox}
  \begin{takeaway}[Mnemonic]
    The diagonal ($855$, $60$) is correct; the off-diagonal ($45$, $40$) are the two error
    types. ``False positive'' names the \emph{predicted} label --- predicted default but
    truly not.
  \end{takeaway}
\end{frame}

\begin{frame}{Problem 3 --- solution (a.2): the five metrics}
  \begin{solutionbox}[Solution 3(a) step 2 --- the five performance metrics]
    \begin{align*}
      \text{accuracy} &= \frac{\text{TP}+\text{TN}}{n}=\frac{60+855}{1000}=\mathbf{0.915}
        \quad(\text{error rate}=0.085),\\
      \text{sensitivity (recall)} &= \frac{\text{TP}}{\text{TP}+\text{FN}}=\frac{60}{100}=\mathbf{0.600},\\
      \text{specificity} &= \frac{\text{TN}}{\text{TN}+\text{FP}}=\frac{855}{900}=\mathbf{0.950},\\
      \text{precision} &= \frac{\text{TP}}{\text{TP}+\text{FP}}=\frac{60}{105}=\mathbf{0.571},\\
      \text{FPR} &= \frac{\text{FP}}{\text{FP}+\text{TN}}=\frac{45}{900}=\mathbf{0.050}=1-\text{specificity}.
    \end{align*}
    \emph{Read-off:} sensitivity detects only $60\,\%$ of the true defaulters; of all flagged
    customers just $57.1\,\%$ (precision) truly default --- the rest are false alarms.
  \end{solutionbox}
\end{frame}

\begin{frame}{Problem 3 --- solution (b): lowering the threshold to 0.2}
  \begin{solutionbox}[Solution 3(b) --- effect on sensitivity and specificity]
    Lowering the cut-off from $0.5$ to $0.2$ flags \emph{more} customers as ``default''
    (predict default whenever $\hat{p}>0.2$):
    \begin{itemize}
      \item Every true defaulter with $0.2<\hat{p}\le 0.5$ is now caught $\Rightarrow$
        FN $\downarrow$, so \textbf{sensitivity increases}.
      \item But more non-defaulters are also flagged $\Rightarrow$ FP $\uparrow$
        (FPR $\uparrow$), so \textbf{specificity decreases}. (The overall error rate
        typically rises, since non-defaulters are the large majority.)
    \end{itemize}
    \textbf{Business reason.} The costs are asymmetric: a missed defaulter (FN --- a credit
    loss) is far more expensive than a false alarm (FP --- e.g.\ an unnecessary credit-line
    review or call). Trading specificity for sensitivity can lower the \emph{expected cost}
    even if the raw error rate goes up.
  \end{solutionbox}
\end{frame}

\begin{frame}{Problem 3 --- solution (c): the ROC curve and AUC}
  \begin{solutionbox}[Solution 3(c) --- what the ROC curve shows]
    The \textbf{ROC curve} plots the \emph{true positive rate} (sensitivity, vertical axis)
    against the \emph{false positive rate} ($1-$specificity, horizontal axis) as the
    classification \emph{threshold} is swept over all values from $1$ down to $0$. Each
    threshold produces one $(\text{FPR},\text{TPR})$ point.
    \begin{itemize}
      \item $\text{AUC}=0.5$: the curve lies on the diagonal --- no better than
        \textbf{random guessing}; the score carries no usable ranking information.
      \item $\text{AUC}=1$: a \textbf{perfect} classifier --- some threshold reaches
        sensitivity $=1$ and specificity $=1$ simultaneously (the curve hugs the top-left
        corner).
    \end{itemize}
    \textbf{Probabilistic reading.} AUC $=$ probability that a randomly chosen defaulter is
    scored higher than a randomly chosen non-defaulter.
  \end{solutionbox}
\end{frame}

\begin{frame}{Problem 4 --- restatement}
  \begin{exercise}[Problem 4 --- Resampling methods (18 P)]
    \begin{itemize}
      \item[(a)] $n=200$: number of model \emph{fits} for the validation-set approach
        (50/50 split) $\cdot$ 10-fold CV $\cdot$ LOOCV --- with justification. \hfill (4 P)
      \item[(b)] LOOCV vs.\ 10-fold CV: bias and variance; which is preferred in
        practice? \hfill (4 P)
      \item[(c)] Show $\Pr(i\in\text{bootstrap sample})=1-(1-1/n)^n$; evaluate at $n=5$;
        limit as $n\to\infty$. \hfill (5 P)
      \item[(d)] $B=3$ bootstrap estimates $\hat{\alpha}^{*}=0.4$ / $0.5$ / $0.6$
        (toy example; in practice $B\approx 1{,}000$): compute
        $\widehat{\operatorname{SE}}_B(\hat{\alpha})$; what does it estimate; why is the
        bootstrap useful? \hfill (5 P)
    \end{itemize}
  \end{exercise}
\end{frame}

\begin{frame}{Problem 4 --- solution (a): number of model fits ($n=200$)}
  \begin{solutionbox}[Solution 4(a) --- how many times is the model refitted?]
    \begin{itemize}
      \item[(i)] \textbf{Validation set (one 50/50 split): $\mathbf{1}$ fit.} Train once on
        the $100$ training observations; evaluate once on the $100$ held-out ones.
      \item[(ii)] \textbf{10-fold CV: $\mathbf{10}$ fits.} Split into $10$ folds of $20$;
        each fold is held out once while the model is refitted on the remaining $180$.
        $10$ folds $\Rightarrow$ $10$ fits.
      \item[(iii)] \textbf{LOOCV: $\mathbf{200}$ fits.} Each single observation is held out
        once and the model refitted on the other $199$; LOOCV $=$ $n$-fold CV, so $n=200$
        fits. (For least squares a shortcut formula avoids the refits, but generically $n$
        fits are needed.)
    \end{itemize}
  \end{solutionbox}
  \begin{takeaway}[Cost ordering]
    Validation set ($1$) $\ll$ 10-fold ($10$) $\ll$ LOOCV ($n=200$): more fits buy a
    lower-bias error estimate at higher computational cost.
  \end{takeaway}
\end{frame}

\begin{frame}{Problem 4 --- solution (b): bias and variance of CV}
  \begin{solutionbox}[Solution 4(b) --- LOOCV vs.\ 10-fold CV]
    \textbf{Bias.} LOOCV trains on $n-1$ observations (almost the full sample) $\Rightarrow$
    \emph{nearly unbiased} for the true test error. 10-fold CV trains on only $0.9\,n$
    observations $\Rightarrow$ it \emph{slightly overestimates} the test error (a little
    more bias).\\[1mm]
    \textbf{Variance.} LOOCV has the \emph{higher} variance: its $n$ fitted models use
    almost identical training sets, so their errors are strongly \emph{positively
    correlated}, and averaging highly correlated quantities reduces variance only a little.
    The $10$ folds overlap less, giving less-correlated fits and a lower-variance estimate.
  \end{solutionbox}
  \begin{takeaway}[Practice]
    Use $k=5$ or $k=10$: a good bias--variance compromise, and far cheaper than the $n$
    refits of LOOCV.
  \end{takeaway}
\end{frame}

\begin{frame}{Problem 4 --- solution (c): bootstrap inclusion probability}
  \begin{solutionbox}[Solution 4(c) --- the inclusion probability]
    A bootstrap sample is $n$ draws \emph{with replacement}. On a single draw, observation
    $i$ is chosen with probability $1/n$ and missed with probability $1-\tfrac{1}{n}$. The
    $n$ draws are independent, so
    \[
      \Pr(i\notin\text{sample})=\Bigl(1-\tfrac{1}{n}\Bigr)^{n}
      \;\Longrightarrow\;
      \Pr(i\in\text{sample})=1-\Bigl(1-\tfrac{1}{n}\Bigr)^{n}.
    \]
    \textbf{At $n=5$:}\quad
    $1-\bigl(\tfrac{4}{5}\bigr)^{5}=1-0.32768=\mathbf{0.672}.$
    \quad\textbf{As $n\to\infty$:}\quad
    $\bigl(1-\tfrac{1}{n}\bigr)^{n}\to e^{-1}$, so
    $1-e^{-1}\approx 1-0.368=\mathbf{0.632}.$
  \end{solutionbox}
  \begin{takeaway}[Rule of thumb]
    A bootstrap sample contains about $2/3$ ($\approx 63.2\,\%$) of the \emph{distinct}
    original observations; the remaining $\approx 37\,\%$ form the ``out-of-bag'' set.
  \end{takeaway}
\end{frame}

\begin{frame}{Problem 4 --- solution (d): bootstrap standard error}
  \begin{solutionbox}[Solution 4(d) --- the bootstrap standard error]
    \textbf{Step 1 --- bootstrap mean.}\quad
    $\bar{\alpha}^{*}=(0.4+0.5+0.6)/3=0.5.$\\[0.5mm]
    \textbf{Step 2 --- deviations.}\quad
    $0.4-0.5=-0.1,\quad 0.5-0.5=0,\quad 0.6-0.5=+0.1.$\\[0.5mm]
    \textbf{Step 3 --- plug into the formula} (with $B-1=2$):
    \[
      \widehat{\operatorname{SE}}_B(\hat{\alpha})
      =\sqrt{\frac{(-0.1)^2+0^2+(0.1)^2}{3-1}}
      =\sqrt{\frac{0.02}{2}}
      =\sqrt{0.01}=\mathbf{0.100}.
    \]
    \textbf{What it estimates.} The standard deviation of $\hat{\alpha}$ over repeated
    samples from the population --- how much the estimate would vary if the data were
    collected anew.
  \end{solutionbox}
  \begin{takeaway}[Why bootstrap]
    It resamples the observed data --- no closed-form SE formula and no new data needed ---
    and works for almost any statistic ($B\approx 1{,}000$ in practice).
  \end{takeaway}
\end{frame}

\begin{frame}{Problem 5 --- restatement}
  \begin{exercise}[Problem 5 --- Model selection with Cp and BIC (12 P)]
    Best subset selection with $n=100$ and $\hat{\sigma}^2=4$ (from the full model):
    \begin{center}
    \begin{tabular}{lcccc}
      \toprule
      $d$ & 1 & 2 & 3 & 4\\
      \midrule
      $\operatorname{RSS}_d$ & 500 & 444 & 428 & 424\\
      \bottomrule
    \end{tabular}
    \end{center}
    \[
    C_p=\tfrac{1}{n}\bigl(\operatorname{RSS}+2d\hat{\sigma}^2\bigr),\qquad
    \operatorname{BIC}=\tfrac{1}{n}\bigl(\operatorname{RSS}+\log(n)\,d\,\hat{\sigma}^2\bigr),
    \qquad \log(100)\approx 4.605 .
    \]
    \begin{itemize}
      \item[(a)] $C_p$ for all four models. \hfill (4 P)
      \item[(b)] BIC for all four models. \hfill (4 P)
      \item[(c)] Using $C_p=(5.080,4.600,4.520,4.560)$ and
        BIC$=(5.184,4.808,4.833,4.977)$: which $d$ does each criterion select? Explain the
        disagreement. \hfill (2 P)
      \item[(d)] Why can training RSS alone never select $d$? \hfill (2 P)
    \end{itemize}
  \end{exercise}
\end{frame}

\begin{frame}{Problem 5 --- solution (a): Mallows' $C_p$}
  \begin{solutionbox}[Solution 5(a) --- the Cp criterion]
    Penalty per predictor: $2\hat{\sigma}^2=2\cdot 4=8$; with $n=100$,
    $C_p=(\operatorname{RSS}_d+8d)/100$. For example,
    \[
      d=1:\ (500+8\cdot 1)/100=508/100=\mathbf{5.080},\qquad
      d=3:\ (428+8\cdot 3)/100=452/100=\mathbf{4.520}.
    \]
    Doing this for every $d$:
    \begin{center}
    \begin{tabular}{lcccc}
      \toprule
      $d$ & 1 & 2 & 3 & 4\\
      \midrule
      penalty $2d\hat{\sigma}^2=8d$ & 8 & 16 & 24 & 32\\
      $\operatorname{RSS}_d+8d$ & 508 & 460 & 452 & 456\\
      $C_p$ & 5.080 & 4.600 & \textbf{4.520} & 4.560\\
      \bottomrule
    \end{tabular}
    \end{center}
    Minimised at $\mathbf{d=3}$ ($C_p=4.520$).
  \end{solutionbox}
\end{frame}

\begin{frame}{Problem 5 --- solution (b): BIC}
  \begin{solutionbox}[Solution 5(b) --- the BIC criterion]
    Penalty per predictor: $\log(n)\,\hat{\sigma}^2=\log(100)\cdot 4=4.605\cdot 4=18.421$;
    thus $\operatorname{BIC}=(\operatorname{RSS}_d+18.421\,d)/100$. For example,
    \[
      d=1:\ (500+18.421\cdot 1)/100=518.421/100=\mathbf{5.184}.
    \]
    Doing this for every $d$ (the totals carry the full precision of $\log 100=4.60517$):
    \begin{center}
    \begin{tabular}{lcccc}
      \toprule
      $d$ & 1 & 2 & 3 & 4\\
      \midrule
      $\operatorname{RSS}_d+18.421\,d$ & 518.421 & 480.841 & 483.262 & 497.683\\
      BIC & 5.184 & \textbf{4.808} & 4.833 & 4.977\\
      \bottomrule
    \end{tabular}
    \end{center}
    Minimised at $\mathbf{d=2}$ ($\operatorname{BIC}=4.808$).
  \end{solutionbox}
\end{frame}

\begin{frame}{Problem 5 --- solution (c): which model, and why they disagree}
  \begin{solutionbox}[Solution 5(c) --- comparing the two penalties]
    From (a) and (b), $C_p=(5.080,\,4.600,\,4.520,\,4.560)$ and
    $\operatorname{BIC}=(5.184,\,4.808,\,4.833,\,4.977)$:
    \[
      C_p \text{ is minimised at } \mathbf{d=3}\ (4.520),
      \qquad
      \operatorname{BIC}\text{ is minimised at } \mathbf{d=2}\ (4.808).
    \]
    Both are ``RSS $+$ size penalty'', but the per-predictor charge differs:
    \[
      C_p:\ 2\hat{\sigma}^2=8
      \qquad\text{vs.}\qquad
      \operatorname{BIC}:\ \log(n)\hat{\sigma}^2=18.421 .
    \]
    Since $\log(100)=4.605>2$, BIC penalises each extra predictor much more heavily and
    prefers \emph{smaller} models. The RSS drop from $d=2$ to $d=3$ is $444-428=16$:
    \emph{larger} than the $C_p$ penalty $8$ (so $C_p$ accepts predictor $3$) but
    \emph{smaller} than the BIC penalty $18.421$ (so BIC rejects it).
  \end{solutionbox}
\end{frame}

\begin{frame}{Problem 5 --- solution (d): why training RSS alone cannot pick $d$}
  \begin{solutionbox}[Solution 5(d) --- RSS is monotone in $d$]
    Adding a predictor to a nested model can never \emph{increase} the training RSS (at
    worst its coefficient is set to $0$). So RSS decreases \emph{monotonically} in $d$ ---
    equivalently, training $R^2$ increases monotonically:
    \[
      500 > 444 > 428 > 424 .
    \]
    RSS-based selection would therefore \emph{always} pick the largest model $d=4$,
    regardless of whether the extra predictors carry real signal: training error is an ever
    more optimistic estimate of test error as flexibility grows.
  \end{solutionbox}
  \begin{takeaway}[The fix]
    Either penalise model size ($C_p$, AIC, BIC, adjusted $R^2$) or estimate the test error
    directly (validation set, cross-validation).
  \end{takeaway}
\end{frame}

\begin{frame}{Problem 6 --- restatement}
  \begin{exercise}[Problem 6 --- Ridge regression and the lasso (15 P)]
    Linear model with standardised predictors $X_1,\dots,X_p$.
    \begin{itemize}
      \item[(a)] Objectives of ridge and lasso; which coefficient is not penalised? \hfill (4 P)
      \item[(b)] Behaviour as $\lambda\to 0$ and $\lambda\to\infty$; which Chapter-3 method

      \item[(c)] Why can the lasso set coefficients exactly to zero (geometry for $p=2$)? \hfill (3 P)

      \item[(e)] Ridge fits at $\lambda=0.1$ and $\lambda=100$ (standardised coefficients)

    \end{itemize}
    \vspace{-4.5mm}
    \begin{center}

      \toprule
       & $\lambda=0.1$ & $\lambda=100$\\
      \midrule
      $\hat{\beta}_1$ & $0.92$ & $0.31$\\
      $\hat{\beta}_2$ & $0.85$ & $0.29$\\
      $\hat{\beta}_3$ & $-0.41$ & $-0.12$\\
      $\hat{\beta}_4$ & $0.20$ & $0.06$\\
      \bottomrule
    \end{tabular}
    \hspace{10mm}
    \begin{tabular}{lcccc}
      \toprule
      $\lambda$ & 0.1 & 1 & 10 & 100\\
      \midrule
      CV error & 12.8 & 11.9 & 10.6 & 13.4\\
      \bottomrule
    \end{tabular}
    \end{center}
  \end{exercise}
\end{frame}

\begin{frame}{Problem 6 --- solution (a): the two objectives}
  \begin{solutionbox}[Solution 6(a) --- ridge vs.\ lasso penalties]
    With
    $\operatorname{RSS}=\sum_{i=1}^{n}\bigl(y_i-\beta_0-\sum_{j=1}^{p}\beta_j x_{ij}\bigr)^2$
    and tuning parameter $\lambda\ge 0$:
    \[
      \text{Ridge: }\ \min_{\beta}\ \operatorname{RSS}+\lambda\sum_{j=1}^{p}\beta_j^{2}
        \;\bigl(=\operatorname{RSS}+\lambda\|\beta\|_2^2\bigr),
    \]
    \[
      \text{Lasso: }\ \min_{\beta}\ \operatorname{RSS}+\lambda\sum_{j=1}^{p}|\beta_j|
        \;\bigl(=\operatorname{RSS}+\lambda\|\beta\|_1\bigr).
    \]
    Ridge uses the squared ($\ell_2$) penalty, the lasso the absolute ($\ell_1$) penalty.
    The intercept $\beta_0$ is \emph{not} penalised --- shrinking the overall level of $y$
    would make no sense.
  \end{solutionbox}
\end{frame}

\begin{frame}{Problem 6 --- solution (b): limiting behaviour of $\lambda$}
  \begin{solutionbox}[Solution 6(b) --- the two extremes]
    \begin{itemize}
      \item $\boldsymbol{\lambda\to 0}$: the penalty vanishes and both estimators approach
        the \emph{ordinary least-squares} (OLS) fit of Chapter~3.
      \item $\boldsymbol{\lambda\to\infty}$: the penalty dominates and all penalised
        coefficients are forced to $0$ --- the null model with intercept only.
    \end{itemize}
    In between, the two behave differently: ridge shrinks coefficients \emph{towards} zero
    (exactly zero only in the limit), whereas the lasso sets more and more coefficients
    \emph{exactly} to zero as $\lambda$ grows. Increasing $\lambda$ traces a whole path of
    fits whose flexibility \emph{decreases} continuously.
  \end{solutionbox}
  \begin{takeaway}[Recovered method]
    As $\lambda\to0$ both penalised methods reduce to \textbf{ordinary least squares}.
  \end{takeaway}
\end{frame}

\begin{frame}{Problem 6 --- solution (c): why the lasso gives exact zeros}
  \begin{solutionbox}[Solution 6(c) --- constraint-region geometry]
    Each penalised problem is equivalent to minimising RSS subject to a \emph{budget}
    constraint on the coefficients:
    \[
      \text{lasso: } |\beta_1|+|\beta_2|\le s\ \ (\text{a \emph{diamond}}),
      \qquad
      \text{ridge: } \beta_1^2+\beta_2^2\le s\ \ (\text{a \emph{disk}}).
    \]
    The RSS has elliptical contours centred at the OLS estimate; the solution is where these
    ellipses first \emph{touch} the constraint region.
    \begin{itemize}
      \item The lasso diamond has sharp \textbf{corners on the axes}. Ellipses typically
        touch a corner --- where one coefficient is exactly $0$ (sparsity / variable
        selection).
      \item The ridge disk has \textbf{no corners}, so the contact point generically has
        \emph{all} coordinates non-zero: ridge shrinks but keeps every predictor.
    \end{itemize}
  \end{solutionbox}
\end{frame}

\begin{frame}{Problem 6 --- solution (d): why standardise the predictors}
  \begin{solutionbox}[Solution 6(d) --- units matter for penalised fits]
    OLS is \emph{scale-equivariant}: rescaling $X_j$ (e.g.\ EUR $\to$ thousand~EUR) simply
    rescales $\hat{\beta}_j$ and leaves the fit unchanged. The penalties, however, depend on
    the coefficient \emph{magnitudes} --- $\lambda\sum\beta_j^2$ resp.\ $\lambda\sum|\beta_j|$
    --- and hence on the units: a predictor measured in small units has a large coefficient
    and would be shrunk more strongly for a purely \emph{cosmetic} reason. Standardising each
    $X_j$ to standard deviation $1$ puts all coefficients on a common scale, so the penalty
    acts comparably on every predictor and the solution is independent of the units.
  \end{solutionbox}
\end{frame}

\begin{frame}{Problem 6 --- solution (e): bias--variance and choosing $\lambda$}
  \begin{solutionbox}[Solution 6(e) --- read the two fits and the CV grid]
    \textbf{Bias--variance of the two fits.}
    $\lambda=0.1$ applies almost no shrinkage --- close to OLS, more flexible $\Rightarrow$
    \textbf{lower bias, higher variance}. $\lambda=100$ shrinks each coefficient to roughly
    a third of its size (e.g.\ $0.92\to0.31$, $0.85\to0.29$; ratios $\approx 0.30$--$0.34$)
    --- a more rigid fit $\Rightarrow$ \textbf{higher bias, lower variance}. This is the
    Chapter-2 trade-off controlled by the single knob $\lambda$.\\[1mm]
    \textbf{Choosing $\lambda$.} Use the smallest \emph{cross-validated} error, not the
    training fit:
    \[
      \lambda=10:\ 10.6 \;<\; 11.9\ (\lambda{=}1) \;<\; 12.8\ (\lambda{=}0.1) \;<\; 13.4\ (\lambda{=}100).
    \]
    So choose $\boldsymbol{\lambda=10}$ --- intermediate shrinkage; \emph{neither} fit in
    the coefficient table is optimal.
  \end{solutionbox}
  \begin{takeaway}[Big picture for the exam]
    Regularisation trades a little bias for a large variance reduction; $\lambda$ is always
    chosen by cross-validation --- never by looking at training RSS.
  \end{takeaway}
\end{frame}
