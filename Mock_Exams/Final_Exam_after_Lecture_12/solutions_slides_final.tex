\begin{frame}{Problem 1 --- task}
  \begin{exercise}[Problem 1 (16 P) --- Moving beyond linearity]
    \begin{itemize}
      \item[(a)] \textbf{[4 P]} Degrees of freedom of a cubic spline with $K$ interior knots ---
        justify by counting parameters and constraints; evaluate for $K=4$.
      \item[(b)] \textbf{[4 P]} Degrees of freedom of a \emph{natural} cubic spline with $K$ knots
        ($K=4$); which extra constraint acts where and why is it useful?
      \item[(c)] \textbf{[4 P]} Smoothing spline
        $\sum_i (y_i-g(x_i))^2+\lambda\int g''(t)^2dt$: role of the two terms;
        behaviour of $\hat g$ as $\lambda\to 0$ and $\lambda\to\infty$.
      \item[(d)] \textbf{[4 P]} Truncated-power basis of a cubic spline with one knot $\xi$;
        define $(x-\xi)^3_+$; why does adding it keep the fit smooth at $\xi$?
    \end{itemize}
  \end{exercise}
  {\scriptsize\itshape Lecture slides: Chapter 7 --- Moving Beyond Linearity (Lecture 9).}
\end{frame}

\begin{frame}{Problem 1 --- solution (a): cubic-spline df}
  \begin{solutionbox}[Solution P1 --- (a) degrees of freedom of a cubic spline]
    \textbf{Strategy: count parameters, then subtract constraints.}
    \begin{itemize}
      \item $K$ interior knots split the range of $X$ into $K+1$ intervals; each interval carries
        its own cubic polynomial with $4$ coefficients:
        \[
          4(K+1)=4K+4\quad\text{free parameters (unconstrained pieces).}
        \]
      \item At \emph{each} knot the neighbouring pieces must agree in value, slope and curvature:
        continuity of $g,\,g',\,g''$ $\Rightarrow$ $3$ constraints per knot, $3K$ in total.
    \end{itemize}
    \[
      \mathrm{df}=\underbrace{4(K+1)}_{\text{parameters}}-\underbrace{3K}_{\text{constraints}}
      =K+4
      \qquad\Longrightarrow\qquad K=4:\ \mathrm{df}=4+4=\mathbf{8}.
    \]
    \textbf{Basis check.} The same space is spanned by $1,\,x,\,x^2,\,x^3$ plus one truncated-power
    term per knot, i.e.\ $4+K$ basis functions --- an identical count.
  \end{solutionbox}
\end{frame}

\begin{frame}{Problem 1 --- solution (b): natural cubic spline}
  \begin{solutionbox}[Solution P1 --- (b) the natural cubic spline]
    A \emph{natural} cubic spline adds one requirement: it must be \emph{linear beyond the two
    boundary knots}, i.e.\ in the outer regions $X<\xi_1$ and $X>\xi_K$.
    \medskip

    \textbf{Cost of the constraint.} Linearity forces the quadratic and cubic terms to vanish in
    each outer region $\Rightarrow$ $2$ fewer parameters per boundary region, $4$ in total:
    \[
      \mathrm{df}=(K+4)-4=K
      \qquad\Longrightarrow\qquad K=4:\ \mathrm{df}=\mathbf{4}.
    \]
    \textbf{Where it acts:} only in the two tails, outside the outermost knots.
  \end{solutionbox}
  \begin{takeaway}[Why it is useful]
    \footnotesize Unconstrained splines and polynomials wiggle wildly near the boundaries, where
    data are sparse; forcing linearity there cuts the prediction \emph{variance} at and beyond the
    boundary for only a little extra bias.
  \end{takeaway}
\end{frame}

\begin{frame}{Problem 1 --- solution (c): smoothing spline}
  \begin{solutionbox}[Solution P1 --- (c) the two terms and the limits of $\lambda$]
    \[
      \min_{g}\ \underbrace{\sum_{i=1}^n\bigl(y_i-g(x_i)\bigr)^2}_{\text{fit to the data (RSS)}}
      \;+\;\lambda\underbrace{\int g''(t)^2\,dt}_{\text{roughness / curvature}} .
    \]
    $g''$ measures curvature, so the integral penalises wiggliness; the tuning parameter
    $\lambda\ge0$ trades the two terms off.
    \begin{itemize}
      \item $\lambda\to 0$: the penalty vanishes $\Rightarrow$ $\hat g$ \emph{interpolates} every
        training point --- maximally wiggly, zero training RSS, severe overfitting
        (effective df $\to n$).
      \item $\lambda\to\infty$: any curvature is infinitely expensive $\Rightarrow g''\equiv0$
        $\Rightarrow$ $\hat g$ is the \emph{least-squares straight line} (effective df $\to2$).
    \end{itemize}
    In between, $\lambda$ (chosen e.g.\ by LOOCV) moves the effective df smoothly from $n$ to $2$.
  \end{solutionbox}
\end{frame}

\begin{frame}{Problem 1 --- solution (d): truncated-power basis}
  \begin{solutionbox}[Solution P1 --- (d) truncated-power basis with one knot $\xi$]
    A cubic spline with a single knot $\xi$ can be written with a five-function basis:
    \[
      f(x)=\beta_0+\beta_1x+\beta_2x^2+\beta_3x^3+\beta_4\,(x-\xi)^3_+,
      \qquad
      (x-\xi)^3_+=\begin{cases}(x-\xi)^3,& x>\xi\\[0.5mm] 0,& x\le\xi.\end{cases}
    \]
    \textbf{Why smoothness survives.} At the knot $x=\xi$ the added function and its first two
    derivatives all vanish:
    \[
      (x-\xi)^3_+\big|_{\xi}=0,\qquad
      \tfrac{d}{dx}(x-\xi)^3_+\big|_{\xi}=0,\qquad
      \tfrac{d^2}{dx^2}(x-\xi)^3_+\big|_{\xi}=0,
    \]
    so adding it changes only the \emph{third} derivative at $\xi$. The fit therefore stays
    continuous with continuous $g'$ and $g''$ --- exactly the defining smoothness of a cubic spline.
  \end{solutionbox}
\end{frame}

\begin{frame}{Problem 2 --- task}
  \begin{exercise}[Problem 2 (8 P) --- Generalised additive models]
    \begin{itemize}
      \item[(a)] \textbf{[3 P]} General form of a GAM for quantitative $Y$; how does it extend
        multiple linear regression?
      \item[(b)] \textbf{[3 P]} One advantage concerning the interpretation of the $f_j$;
        the main structural limitation --- and its repair.
      \item[(c)] \textbf{[2 P]} Backfitting in one or two sentences.
    \end{itemize}
  \end{exercise}
  {\scriptsize\itshape Lecture slides: Chapter 7 --- Moving Beyond Linearity (Lecture 9).}
\end{frame}

\begin{frame}{Problem 2 --- solution (a): the GAM model form}
  \begin{solutionbox}[Solution P2 --- (a) general form of a GAM]
    \[
      Y=\beta_0+f_1(X_1)+f_2(X_2)+\dots+f_p(X_p)+\varepsilon .
    \]
    Each predictor enters through its \emph{own} smooth function $f_j$ (a natural or smoothing
    spline, or local regression), and the contributions are \emph{added}.
    \medskip

    \textbf{Extension of linear regression.} Multiple linear regression is the special case
    $f_j(X_j)=\beta_jX_j$; a GAM keeps the additive structure but frees each term to bend into a
    smooth non-linearity --- capturing curved effects a straight line would miss.
  \end{solutionbox}
  \begin{takeaway}[One step further]
    \footnotesize The same idea extends to classification: place the additive predictor inside a
    link, e.g.\ $\log\!\frac{p(X)}{1-p(X)}=\beta_0+\sum_j f_j(X_j)$ for a logistic GAM.
  \end{takeaway}
\end{frame}

\begin{frame}{Problem 2 --- solution (b): advantage and limitation}
  \begin{solutionbox}[Solution P2 --- (b) interpretability vs.\ interactions]
    \textbf{Advantage (interpretability).} Because the model is additive, the effect of each
    $X_j$ on $Y$ is captured by its own function $f_j$, which can be \emph{plotted and read one
    panel at a time} while the other predictors are held fixed --- non-linear effects stay as
    readable as a coefficient table.
    \medskip

    \textbf{Limitation.} Additivity excludes \emph{interactions}: the effect of $X_j$ cannot
    depend on the value of $X_k$.
    \medskip

    \textbf{Repair.} Add interaction components by hand --- a product term $X_j\times X_k$, or a
    low-dimensional joint smooth $f_{jk}(X_j,X_k)$ as an extra component.
  \end{solutionbox}
\end{frame}

\begin{frame}{Problem 2 --- solution (c): backfitting}
  \begin{solutionbox}[Solution P2 --- (c) the backfitting algorithm]
    Fit the $f_j$ one at a time, cycling through $j=1,\dots,p$. At each step update $f_j$ by
    smoothing the \emph{partial residuals} --- what remains after removing all the \emph{other}
    current terms:
    \[
      r_i=y_i-\beta_0-\sum_{k\ne j} f_k(x_{ik}),\qquad\text{then re-fit } f_j \text{ to } (x_{ij},r_i).
    \]
    Every other function is held fixed while $f_j$ is updated; repeat the whole cycle until the
    $f_j$ stop changing (convergence).
  \end{solutionbox}
  \begin{takeaway}[Why it is convenient]
    \footnotesize Backfitting only ever needs a \emph{one-dimensional} smoother, so any
    scatterplot smoother (spline, loess) can be plugged in --- no giant multivariate fit required.
  \end{takeaway}
\end{frame}

\begin{frame}{Problem 3 --- task}
  \begin{exercise}[Problem 3 (20 P) --- Trees by hand]
    Advertising budget $x$ (1{,}000 EUR) and weekly sales $y$ (1{,}000 units) for $n=6$ products;
    one split at $s$: $R_1=\{x<s\}$ and $R_2=\{x\ge s\}$ with region-mean predictions.
    \begin{center}\scriptsize
    \begin{tabular}{lcccccc}
      \toprule
      $i$ & 1 & 2 & 3 & 4 & 5 & 6\\
      \midrule
      $x_i$ & 1 & 2 & 3 & 4 & 5 & 6\\
      $y_i$ & 2 & 4 & 3 & 3 & 8 & 10\\
      \bottomrule
    \end{tabular}
    \end{center}
    (a) \textbf{[8 P]} Evaluate the cutpoints $s=2.5$ and $s=4.5$ by total RSS; which is chosen?
    (b) \textbf{[2 P]} Using the chosen split $s=4.5$ from (a) ($R_1=\{x<4.5\}$, $\hat y_{R_1}=3$;
        $R_2=\{x\ge 4.5\}$, $\hat y_{R_2}=9$), prediction for a new product with $x=5$.
    (c) \textbf{[6 P]} Node with 6 ``Yes'' / 2 ``No'': proportions; Gini; entropy
        ($\ln 0.75=-0.2877$; $\ln 0.25=-1.3863$); predicted class.
    (d) \textbf{[4 P]} Cost-complexity pruning: role of $\alpha$ in
        $\sum_m\sum_{i\in R_m}(y_i-\hat y_{R_m})^2+\alpha|T|$.
  \end{exercise}
  {\scriptsize\itshape Lecture slides: Chapter 8 --- Tree-Based Methods (Lecture 10).}
\end{frame}

\begin{frame}{Problem 3 --- solution (a.1): root node and cutpoint $s=2.5$}
  \begin{solutionbox}[Solution P3 --- (a) part 1: root node and cutpoint 2.5]
    \textbf{Root node (no split), for reference.}
    $\bar y=\tfrac{2+4+3+3+8+10}{6}=\tfrac{30}{6}=5$, and
    \[
      \mathrm{RSS}_{\text{root}}=(2{-}5)^2+(4{-}5)^2+(3{-}5)^2+(3{-}5)^2+(8{-}5)^2+(10{-}5)^2
      =9{+}1{+}4{+}4{+}9{+}25=52 .
    \]
    \textbf{Cutpoint $s=2.5$:} $R_1=\{1,2\}$, $y\in\{2,4\}$, $\hat y_{R_1}=3$;\quad
    $R_2=\{3,4,5,6\}$, $y\in\{3,3,8,10\}$, $\hat y_{R_2}=\tfrac{24}{4}=6$.
    \[
      \mathrm{RSS}_{R_1}=(2{-}3)^2+(4{-}3)^2=2,
    \]
    \[
      \mathrm{RSS}_{R_2}=(3{-}6)^2+(3{-}6)^2+(8{-}6)^2+(10{-}6)^2=9{+}9{+}4{+}16=38,
    \]
    \[
      \Longrightarrow\quad \mathrm{RSS}(2.5)=2+38=\mathbf{40}.
    \]
  \end{solutionbox}
\end{frame}

\begin{frame}{Problem 3 --- solution (a.2): cutpoint $s=4.5$ and the choice}
  \begin{solutionbox}[Solution P3 --- (a) part 2: cutpoint 4.5 and the choice]
    \textbf{Cutpoint $s=4.5$:} $R_1=\{1,2,3,4\}$, $y\in\{2,4,3,3\}$, $\hat y_{R_1}=\tfrac{12}{4}=3$;\quad
    $R_2=\{5,6\}$, $y\in\{8,10\}$, $\hat y_{R_2}=9$.
    \[
      \mathrm{RSS}_{R_1}=1+1+0+0=2,\qquad
      \mathrm{RSS}_{R_2}=(8{-}9)^2+(10{-}9)^2=2
      \ \Rightarrow\ \mathrm{RSS}(4.5)=2+2=\mathbf{4}.
    \]
    \begin{center}\scriptsize
    \begin{tabular}{lccccc}
      \toprule
      $s$ & $R_1$ ($y$) & $\hat y_{R_1}$ & $R_2$ ($y$) & $\hat y_{R_2}$ & total RSS\\
      \midrule
      $2.5$ & $\{2,4\}$ & $3$ & $\{3,3,8,10\}$ & $6$ & $2+38=40$\\
      $4.5$ & $\{2,4,3,3\}$ & $3$ & $\{8,10\}$ & $9$ & $2+2=\mathbf{4}$\\
      \bottomrule
    \end{tabular}
    \end{center}
    \textbf{Decision.} Recursive binary splitting is greedy and picks the smaller RSS: $4<40$, so
    it chooses $\mathbf{s=4.5}$ --- isolating the four low-sales products (mean $3$) from the two
    high-sales products (mean $9$) and cutting the root RSS from $52$ down to $4$.
  \end{solutionbox}
\end{frame}

\begin{frame}{Problem 3 --- solution (b): prediction for a new product}
  \begin{solutionbox}[Solution P3 --- (b) prediction for a new product]
    Drop the new observation down the tree using the chosen split $s=4.5$:
    \[
      x=5\ \ge\ 4.5 \quad\Longrightarrow\quad \text{region } R_2
      \quad\Longrightarrow\quad \hat y=\hat y_{R_2}=\mathbf{9}
    \]
    i.e.\ the tree predicts $\mathbf{9{,}000}$ units per week.
  \end{solutionbox}
  \begin{takeaway}[How a tree predicts]
    \footnotesize A regression tree is \emph{piecewise constant}: every $x\in[4.5,6]$ receives the
    same prediction $\hat y_{R_2}=9$ (the region mean). Predictions change only by jumping at the
    split points.
  \end{takeaway}
\end{frame}

\begin{frame}{Problem 3 --- solution (c): Gini and entropy of a node}
  \begin{solutionbox}[Solution P3 --- (c) impurity of a 6-Yes / 2-No node]
    Class proportions: $\hat p_{\text{Yes}}=\tfrac{6}{8}=0.75$,\quad $\hat p_{\text{No}}=\tfrac{2}{8}=0.25$.
    \[
      G=\sum_k \hat p_k(1-\hat p_k)=0.75\cdot0.25+0.25\cdot0.75=0.1875+0.1875=\mathbf{0.375}.
    \]
    \[
      D=-\sum_k \hat p_k\ln\hat p_k=-\bigl(0.75\ln0.75+0.25\ln0.25\bigr)
    \]
    \[
      =-\bigl(0.75\cdot(-0.2877)+0.25\cdot(-1.3863)\bigr)=0.2158+0.3466=\mathbf{0.562}.
    \]
    \textbf{Predicted class:} the majority, \textbf{Yes} (estimated probability $0.75$).
  \end{solutionbox}
  \begin{takeaway}[Reading the numbers]
    \footnotesize Both $G$ and $D$ equal $0$ for a pure node and are largest at a $50\!:\!50$ split;
    here they sit in the middle, so this node is \emph{moderately impure}.
  \end{takeaway}
\end{frame}

\begin{frame}{Problem 3 --- solution (d): cost-complexity pruning}
  \begin{solutionbox}[Solution P3 --- (d) cost-complexity pruning]
    In $\ \sum_{m}\sum_{i\in R_m}(y_i-\hat y_{R_m})^2+\alpha|T|\ $ the parameter $\alpha\ge0$ is the
    \emph{price charged per terminal node} $|T|$: it trades the subtree's fit against its size.
    \begin{itemize}
      \item $\alpha=0$: the criterion is just the training RSS $\Rightarrow$ the full tree $T_0$
        wins (nothing penalises size).
      \item $\alpha$ increasing: each leaf must ``pay for itself''; a branch whose RSS reduction is
        smaller than $\alpha$ is pruned. Growing $\alpha$ yields a \emph{nested sequence} of
        ever-smaller subtrees (weakest-link pruning).
      \item Choosing $\alpha$: by \emph{cross-validation} --- take the $\alpha$ with the smallest
        CV error, then refit that subtree on all the data.
    \end{itemize}
  \end{solutionbox}
  \begin{takeaway}[Bias--variance]
    \footnotesize $\alpha$ is a complexity dial: small $\alpha$ = large low-bias/high-variance
    tree; large $\alpha$ = small high-bias/low-variance tree. CV locates the sweet spot.
  \end{takeaway}
\end{frame}

\begin{frame}{Problem 4 --- task}
  \begin{exercise}[Problem 4 (16 P) --- Ensembles]
    \begin{itemize}
      \item[(a)] \textbf{[6 P]} $B$ identically distributed trees with variance $\sigma^2$ and
        pairwise correlation $\rho$:
        $\operatorname{Var}\bigl(\frac1B\sum_b\hat f_b(x)\bigr)=\rho\sigma^2+\frac{1-\rho}{B}\sigma^2$.
        (i) Why does this show variance reduction and what limits it?
        (ii) Evaluate for $B=100$ and $\rho=0.6$ and $\sigma^2=4$; compare with one tree.
        (iii) Limit as $B\to\infty$?
      \item[(b)] \textbf{[4 P]} Random forests: modification at each split; recommended $m$
        for $p=25$; why does it improve on bagging (use
        $\operatorname{Var}=\rho\sigma^2+\frac{1-\rho}{B}\sigma^2$ from (a))?
      \item[(c)] \textbf{[4 P]} Boosting via
        \texttt{GradientBoostingRegressor(n\_estimators=5000,} \texttt{learning\_rate=0.01,}
        \texttt{max\_depth=1)}: what does each of the three arguments control? The
        slow-learning idea (and how the chosen values reflect it)?
      \item[(d)] \textbf{[2 P]} What does the OOB error estimate and why is it free?
    \end{itemize}
  \end{exercise}
  {\scriptsize\itshape Lecture slides: Chapter 8 --- Tree-Based Methods (Lecture 10).}
\end{frame}

\begin{frame}{Problem 4 --- solution (a.1): why bagging reduces variance}
  \begin{solutionbox}[Solution P4 --- (a)(i) reading the variance formula]
    \[
      \operatorname{Var}\!\Bigl(\tfrac1B\textstyle\sum_{b=1}^B \hat f_b(x)\Bigr)
      =\underbrace{\rho\,\sigma^2}_{\text{correlated part}}
      +\underbrace{\frac{1-\rho}{B}\,\sigma^2}_{\text{independent part}} .
    \]
    \begin{itemize}
      \item The second term $\frac{1-\rho}{B}\sigma^2\to0$ as $B$ grows: averaging cancels the
        \emph{independent} part of the trees' variability --- this is the variance reduction of
        bagging.
      \item The first term $\rho\sigma^2$ does \emph{not} depend on $B$. Bootstrap trees are grown
        on overlapping samples of the same data and share the same strong predictors, so they are
        positively correlated; that correlated part cannot be averaged away and \emph{limits} the
        reduction.
    \end{itemize}
  \end{solutionbox}
\end{frame}

\begin{frame}{Problem 4 --- solution (a.2): evaluate and take $B\to\infty$}
  \begin{solutionbox}[Solution P4 --- (a)(ii)--(iii) the numbers and the limit]
    \textbf{(ii) $B=100$, $\rho=0.6$, $\sigma^2=4$:}
    \[
      \operatorname{Var}=0.6\cdot4+\frac{1-0.6}{100}\cdot4
      =2.4+\frac{0.4\cdot4}{100}=2.4+0.016=\mathbf{2.416},
    \]
    versus $\sigma^2=4$ for a single tree --- a reduction of about $\mathbf{40\%}$.
    \medskip

    \textbf{(iii) $B\to\infty$:} the independent term dies and the variance approaches the floor
    \[
      \operatorname{Var}\ \longrightarrow\ \rho\sigma^2=0.6\cdot4=\mathbf{2.4}.
    \]
  \end{solutionbox}
  \begin{takeaway}[More trees never overfit]
    \footnotesize Raising $B$ can only push the variance from $2.416$ down toward $2.4$; extra
    trees merely stabilise the average. To go \emph{below} $2.4$ you must lower $\rho$ --- exactly
    what random forests do.
  \end{takeaway}
\end{frame}

\begin{frame}{Problem 4 --- solution (b): random forests}
  \begin{solutionbox}[Solution P4 --- (b) decorrelating the trees]
    \textbf{Modification.} At \emph{each split}, only a fresh \emph{random subset} of $m$ of the
    $p$ predictors is allowed as splitting candidates.
    \medskip

    \textbf{Recommended $m$ (classification):} $m\approx\sqrt p$, so for $p=25$:
    $m=\sqrt{25}=\mathbf{5}$.
    \medskip

    \textbf{Why it beats bagging.} With bagging a dominant predictor tops the first split of
    almost every tree $\Rightarrow$ the trees look alike $\Rightarrow$ $\rho$ is high. Withholding
    it from $\approx\tfrac{p-m}{p}$ of the split searches \emph{decorrelates} the trees, lowering
    $\rho$ --- and by (a) a smaller $\rho$ lowers the irreducible floor $\rho\sigma^2$ that
    averaging alone can never touch.
  \end{solutionbox}
\end{frame}

\begin{frame}{Problem 4 --- solution (c.1): the three boosting parameters}
  \begin{solutionbox}[Solution P4 --- (c) what each constructor argument controls]
    \begin{center}\ttfamily\scriptsize
    GradientBoostingRegressor(n\_estimators=5000,\\
    \ \ learning\_rate=0.01, max\_depth=1)
    \end{center}
    \begin{itemize}
      \item \texttt{n\_estimators}$=B$ --- \textbf{number of trees} added sequentially; unlike
        bagging, boosting \emph{can} overfit if $B$ is too large, so choose $B$ by CV (here
        $5000$).
      \item \texttt{learning\_rate}$=\lambda$ --- \textbf{shrinkage}: scales each new tree's
        contribution, i.e.\ the speed of learning; smaller $\lambda$ needs larger $B$ (here
        $0.01$).
      \item \texttt{max\_depth}$=d$ --- \textbf{splits per tree} (interaction depth): the
        complexity and interaction order of each tree; often $d=1$ stumps suffice (here $1$).
    \end{itemize}
  \end{solutionbox}
\end{frame}

\begin{frame}{Problem 4 --- solution (c.2): the ``slow learning'' idea}
  \begin{solutionbox}[Solution P4 --- (c) why boosting learns slowly]
    Boosting builds the model \emph{sequentially}: each new tree is fitted to the current
    \emph{residuals} (what the model still gets wrong), and only the small fraction $\lambda$ of
    that tree is added,
    \[
      \hat f(x)\ \leftarrow\ \hat f(x)+\lambda\,\hat f^{\text{new}}(x).
    \]
    So the fit improves gradually and concentrates effort exactly where errors remain, instead of
    fitting the data hard in one step --- which tends to generalise better.
    \medskip

    \textbf{The chosen values encode this:} a \emph{small} \texttt{learning\_rate}$=0.01$ with
    \emph{many} trees (\texttt{n\_estimators}$=5000$) and shallow \texttt{max\_depth}$=1$ stumps.
  \end{solutionbox}
\end{frame}

\begin{frame}{Problem 4 --- solution (d): out-of-bag error}
  \begin{solutionbox}[Solution P4 --- (d) what OOB estimates and why it is free]
    Each bootstrap sample is drawn with replacement, so on average about $\tfrac13$ of the
    observations are left out --- they are \emph{out-of-bag} (OOB) for that tree.
    \medskip

    Predict each observation using \emph{only} the trees that never saw it, then aggregate: the
    resulting error is the \textbf{OOB error}, a valid estimate of the model's \emph{test error}.
    \medskip

    \textbf{Why free:} it reuses the bootstrap as a by-product --- no separate validation set and
    no extra cross-validation loop are required.
  \end{solutionbox}
  \begin{takeaway}[Handy consequence]
    \footnotesize For large $B$ the OOB error is essentially the same as LOOCV, but obtained in a
    single training run.
  \end{takeaway}
\end{frame}

\begin{frame}{Problem 5 --- task}
  \begin{exercise}[Problem 5 (20 P) --- Neural networks by hand]
    Input $x=(1{,}2)$; hidden ReLU units $g(z)=\max\{0{,}z\}$; linear output:
    \[
      A_1=g(-1+1x_1+2x_2)\qquad A_2=g(-3-2x_1+1x_2)\qquad f(x)=1+2A_1+4A_2
    \]
    \begin{itemize}
      \item[(a)] \textbf{[8 P]} Forward pass step by step: pre-activations; $A_1$ and $A_2$;
        output $f(x)$. What did the ReLU do to unit 2 and why are non-linear activations essential?
      \item[(b)] \textbf{[6 P]} Logits $z=(1.0{,}\ 0.0{,}\ -1.0)$; true class 1: all three softmax
        probabilities and the cross-entropy loss
        ($e^{1}=2.718$; $e^{0}=1.000$; $e^{-1}=0.368$; $\ln 4.086=1.408$).
      \item[(c)] \textbf{[4 P]} Parameter count of a $p=4\to k=3\to K=2$ network with biases
        (show both layers).
      \item[(d)] \textbf{[2 P]} Conv layer: $32\times 32$ input; $F=5$; $P=2$; $S=1$:
        output size via $(W-F+2P)/S+1$.
    \end{itemize}
  \end{exercise}
  {\scriptsize\itshape Lecture slides: Chapter 10 --- Deep Learning (Lecture 11).}
\end{frame}

\begin{frame}{Problem 5 --- solution (a): forward pass}
  \begin{solutionbox}[Solution P5 --- (a) the forward pass]
    \textbf{Pre-activations} (weighted sum $+$ bias):
    \[
      z_1=-1+1\cdot1+2\cdot2=-1+1+4=4,\qquad
      z_2=-3-2\cdot1+1\cdot2=-3-2+2=-3 .
    \]
    \textbf{ReLU activations} $g(z)=\max\{0,z\}$:
    \[
      A_1=\max\{0,4\}=4,\qquad A_2=\max\{0,-3\}=0 .
    \]
    \textbf{Output} (linear unit):
    \[
      f(x)=1+2A_1+4A_2=1+2\cdot4+4\cdot0=\mathbf{9}.
    \]
  \end{solutionbox}
  \begin{takeaway}[What the ReLU did / why it matters]
    \footnotesize Unit 2 is \emph{switched off} ($z_2<0\Rightarrow A_2=0$) and adds nothing here
    (it could for other inputs). Without a non-linear $g$ the network collapses into one linear
    function of $x$ --- the non-linearity is what buys flexibility.
  \end{takeaway}
\end{frame}

\begin{frame}{Problem 5 --- solution (b): softmax and cross-entropy}
  \begin{solutionbox}[Solution P5 --- (b) softmax and cross-entropy]
    \abovedisplayskip=4pt\belowdisplayskip=4pt\abovedisplayshortskip=2pt\belowdisplayshortskip=2pt
    \textbf{Exponentiate the logits and sum:}
    \[
      e^{1.0}=2.718,\quad e^{0.0}=1.000,\quad e^{-1.0}=0.368
      \ \Rightarrow\ \textstyle\sum=2.718+1.000+0.368=4.086 .
    \]
    \textbf{Softmax probabilities:}
    \[
      \hat p_1=\tfrac{2.718}{4.086}=\mathbf{0.665},\quad
      \hat p_2=\tfrac{1.000}{4.086}=\mathbf{0.245},\quad
      \hat p_3=\tfrac{0.368}{4.086}=\mathbf{0.090}
      \quad(\text{sum}=1.000\ \checkmark).
    \]
    \textbf{Cross-entropy for the true class 1:}
    \[
      -\ln\hat p_1=-\ln\frac{e^{1}}{4.086}=-(1-\ln4.086)=1.408-1=\mathbf{0.408}.
    \]
  \end{solutionbox}
  \begin{takeaway}[Sanity check]
    \footnotesize A perfect $\hat p_1=1$ gives loss $0$; the uniform guess $\tfrac13$ gives
    $\ln3\approx1.099$. Our $0.408$ sits well below --- the net is right and fairly confident.
  \end{takeaway}
\end{frame}

\begin{frame}{Problem 5 --- solution (c): parameter count}
  \begin{solutionbox}[Solution P5 --- (c) parameters of a $4\to3\to2$ network]
    Every hidden and output unit has a bias, so each layer contributes
    (inputs $\times$ units) weights plus (units) biases:
    \begin{center}
    \begin{tabular}{@{}lccc@{}}
      \toprule
      Layer & Weights & Biases & Subtotal\\
      \midrule
      input $\to$ hidden & $4\cdot3=12$ & $3$ & $15$\\
      hidden $\to$ output & $3\cdot2=6$ & $2$ & $8$\\
      \midrule
      \textbf{Total} & & & $\mathbf{23}$\\
      \bottomrule
    \end{tabular}
    \end{center}
  \end{solutionbox}
  \begin{takeaway}[General rule]
    \footnotesize A dense layer from $a$ to $b$ units has $a\cdot b+b=b(a{+}1)$ parameters. Summing
    the two layers: $3(4{+}1)+2(3{+}1)=15+8=23$.
  \end{takeaway}
\end{frame}

\begin{frame}{Problem 5 --- solution (d): convolution output size}
  \begin{solutionbox}[Solution P5 --- (d) output of a convolutional layer]
    With input width $W=32$, filter size $F=5$, padding $P=2$ and stride $S=1$:
    \[
      \frac{W-F+2P}{S}+1=\frac{32-5+2\cdot2}{1}+1=\frac{31}{1}+1=32 .
    \]
    The height is computed identically, so the output feature map is $\mathbf{32\times32}$
    (one such map per filter).
  \end{solutionbox}
  \begin{takeaway}[``Same'' convolution]
    \footnotesize Here $P=(F-1)/2$ with $S=1$, which preserves the spatial size --- a ``same''
    convolution. A larger stride $S>1$ or less padding would shrink the feature map.
  \end{takeaway}
\end{frame}

\begin{frame}{Problem 6 --- task}
  \begin{exercise}[Problem 6 (20 P) --- Multiple testing]
    \begin{itemize}
      \item[(a)] \textbf{[4 P]} $m=20$ independent tests at $\alpha=0.05$; all nulls true:
        define and compute the FWER ($(0.95)^{20}=0.3585$); interpret.
      \item[(b)] \textbf{[7 P]} Six ordered $p$-values
        $0.001$; $0.004$; $0.011$; $0.030$; $0.045$; $0.230$ --- FWER control at $\alpha=0.05$
        with (i) Bonferroni and (ii) Holm; thresholds; rejected hypotheses; why is Holm never
        weaker than Bonferroni?
      \item[(c)] \textbf{[6 P]} Benjamini--Hochberg at $q=0.05$ on the same six $p$-values from (b)
        ($0.001$; $0.004$; $0.011$; $0.030$; $0.045$; $0.230$):
        thresholds $jq/m$; rejections; what exactly does FDR control guarantee?
      \item[(d)] \textbf{[3 P]} Exploratory screen of 20{,}000 genes with follow-up validation:
        FWER or FDR? Justify.
    \end{itemize}
  \end{exercise}
  {\scriptsize\itshape Lecture slides: Chapter 13 --- Multiple Testing (Lecture 12).}
\end{frame}

\begin{frame}{Problem 6 --- solution (a): FWER of 20 unadjusted tests}
  \begin{solutionbox}[Solution P6 --- (a) family-wise error rate]
    \textbf{Definition.} $\text{FWER}=\Pr(V\ge1)$, the probability of \emph{at least one} false
    rejection, where $V$ counts the true nulls that are wrongly rejected.
    \medskip

    \textbf{Compute} (all $20$ nulls true, tests independent, each at level $0.05$):
    \[
      \text{FWER}=1-\Pr(\text{no false rejection})=1-(1-0.05)^{20}=1-(0.95)^{20}
      =1-0.3585=\mathbf{0.6415}.
    \]
  \end{solutionbox}
  \begin{takeaway}[Interpretation]
    \footnotesize Each test errs only $5\%$ of the time, yet there is a $\approx64\%$ chance of at
    least one spurious ``discovery'' among the $20$. With $m=100$ it is
    $1-0.95^{100}\approx0.994$ --- unadjusted multiple testing all but guarantees false findings.
  \end{takeaway}
\end{frame}

\begin{frame}{Problem 6 --- solution (b.1): Bonferroni}
  \begin{solutionbox}[Solution P6 --- (b) the Bonferroni correction]
    \textbf{Rule.} Reject $H_{(j)}$ iff $p_{(j)}\le\dfrac{\alpha}{m}=\dfrac{0.05}{6}=0.00833$
    --- the \emph{same} threshold for every $j$.
    \medskip

    \begin{center}\scriptsize
    \begin{tabular}{lcccccc}
      \toprule
      $j$ & 1 & 2 & 3 & 4 & 5 & 6\\
      \midrule
      $p_{(j)}$ & 0.001 & 0.004 & 0.011 & 0.030 & 0.045 & 0.230\\
      $\le 0.00833$? & \checkmark & \checkmark & $\times$ & $\times$ & $\times$ & $\times$\\
      \bottomrule
    \end{tabular}
    \end{center}
    \medskip
    Only $0.001$ and $0.004$ clear the bar ($0.011>0.00833$) $\Rightarrow$ reject $H_{(1)}$ and
    $H_{(2)}$: \textbf{2 rejections}.
  \end{solutionbox}
\end{frame}

\begin{frame}{Problem 6 --- solution (b.2): Holm step-down}
  \begin{solutionbox}[Solution P6 --- (b) the Holm step-down procedure]
    \textbf{Rule.} Step down $j=1,2,\dots$, comparing $p_{(j)}$ with $\dfrac{\alpha}{m-j+1}$;
    reject while $p_{(j)}$ stays below its threshold and \emph{stop at the first failure}.
    \begin{center}\scriptsize
    \begin{tabular}{lcccccc}
      \toprule
      $j$ & 1 & 2 & 3 & 4 & 5 & 6\\
      \midrule
      $p_{(j)}$ & 0.001 & 0.004 & 0.011 & 0.030 & 0.045 & 0.230\\
      $\alpha/(m-j+1)$ & 0.00833 & 0.0100 & 0.0125 & 0.0167 & 0.0250 & 0.0500\\
      decision & \checkmark & \checkmark & \checkmark & \textbf{stop} & --- & ---\\
      \bottomrule
    \end{tabular}
    \end{center}
    $0.011\le0.0125$ but $0.030>0.0167$ $\Rightarrow$ reject $H_{(1)},H_{(2)},H_{(3)}$:
    \textbf{3 rejections}.
  \end{solutionbox}
  \begin{takeaway}[Holm $\supseteq$ Bonferroni]
    \footnotesize Holm's hurdles satisfy $\alpha/(m-j+1)\ge\alpha/m$ for every $j$ (equal only at
    $j=1$), so Holm rejects \emph{at least} whatever Bonferroni does --- uniformly more powerful,
    and still valid without any independence assumption.
  \end{takeaway}
\end{frame}

\begin{frame}{Problem 6 --- solution (c.1): Benjamini--Hochberg}
  \begin{solutionbox}[Solution P6 --- (c) the Benjamini--Hochberg procedure]
    \textbf{Rule.} Compare $p_{(j)}$ with $\dfrac{jq}{m}=\dfrac{j\cdot0.05}{6}$; let $L$ be the
    \emph{largest} $j$ that passes, then reject $H_{(1)},\dots,H_{(L)}$.
    \begin{center}\scriptsize
    \begin{tabular}{lcccccc}
      \toprule
      $j$ & 1 & 2 & 3 & 4 & 5 & 6\\
      \midrule
      $p_{(j)}$ & 0.001 & 0.004 & 0.011 & 0.030 & 0.045 & 0.230\\
      $jq/m$ & 0.00833 & 0.0167 & 0.0250 & 0.0333 & 0.0417 & 0.0500\\
      $p_{(j)}\le jq/m$? & yes & yes & yes & \textbf{yes} & no & no\\
      \bottomrule
    \end{tabular}
    \end{center}
    Largest passing index: $L=4$ ($0.030\le0.0333$, while $0.045>0.0417$). BH rejects \emph{all}
    of $H_{(1)},\dots,H_{(4)}$: \textbf{4 rejections} --- note $H_{(4)}$ is rejected even though on
    its own $0.030$ is not tiny.
  \end{solutionbox}
\end{frame}

\begin{frame}{Problem 6 --- solution (c.2): what FDR control guarantees}
  \begin{solutionbox}[Solution P6 --- (c) the FDR guarantee]
    \[
      \text{FDR}=\mathbb{E}\!\left[\frac{V}{\max(R,1)}\right]\le q=0.05,
    \]
    where $R$ is the number of rejections and $V$ the number of \emph{false} ones. In words:
    \emph{on average} at most $5\%$ of the hypotheses BH rejects are false discoveries.
    \medskip

    It does \emph{not} promise that at most $5\%$ of \emph{these particular} $4$ rejections are
    false, nor does it bound $\Pr(V\ge1)$ --- that stronger guarantee would be FWER control.
  \end{solutionbox}
  \begin{takeaway}[Same data --- three procedures]
    \footnotesize Bonferroni $\to2$, Holm $\to3$, BH $\to4$ rejections. Controlling the milder FDR
    instead of the FWER buys more power (more discoveries) at the cost of a few expected false
    positives.
  \end{takeaway}
\end{frame}

\begin{frame}{Problem 6 --- solution (d): FWER or FDR for a genomics screen?}
  \begin{solutionbox}[Solution P6 --- (d) exploratory screen of 20000 genes]
    \textbf{Recommendation: control the FDR.}
    \begin{itemize}
      \item FWER control (e.g.\ Bonferroni at $\dfrac{0.05}{20{,}000}=2.5\times10^{-6}$) guards
        against even a \emph{single} false positive $\Rightarrow$ almost no power; nearly every
        true signal would be missed.
      \item The study is \emph{exploratory} and every hit is re-tested in an independent follow-up,
        so a few false leads are cheap --- what matters is that the candidate \emph{list} is mostly
        reliable.
      \item FDR control at $q=0.05$--$0.2$ delivers exactly that: a rich list of candidates of
        which, on average, only a known small fraction is false.
    \end{itemize}
  \end{solutionbox}
\end{frame}

\begin{frame}{Problem 7 --- solution (a): flexibility and test error}
  \begin{solutionbox}[Solution P7 --- (a) ``more flexible $\Rightarrow$ lower test error''?]
    \textbf{False.} The expected test error decomposes as
    \[
      \mathbb{E}\bigl(y_0-\hat f(x_0)\bigr)^2
      =\underbrace{\operatorname{Var}(\hat f(x_0))}_{\uparrow\text{ with flexibility}}
      +\underbrace{[\operatorname{Bias}(\hat f(x_0))]^2}_{\downarrow\text{ with flexibility}}
      +\underbrace{\operatorname{Var}(\varepsilon)}_{\text{irreducible}} .
    \]
    Flexibility lowers bias but raises variance; past the sweet spot the variance increase wins and
    the test error \emph{rises} --- the characteristic U-shape. E.g.\ when the true $f$ is nearly
    linear or $n$ is small, a flexible method overfits and a simple linear fit has \emph{lower}
    test error.
  \end{solutionbox}
  \begin{takeaway}[The one thing that is monotone]
    \footnotesize Only the \emph{training} error falls monotonically with flexibility; test error
    is U-shaped --- which is exactly why we \emph{estimate} it (CV) rather than trust training fit.
  \end{takeaway}
\end{frame}

\begin{frame}{Problem 7 --- solution (b): odds vs.\ probability}
  \begin{solutionbox}[Solution P7 --- (b) interpreting the logistic coefficient $0.7$]
    From \texttt{sm.Logit}, $\hat\beta_1=0.7$ for \texttt{missed\_payments} (highly significant,
    $z=4.667$). One more missed payment adds $0.7$ to the \emph{log-odds}, i.e.\ \emph{multiplies
    the odds} of default by
    \[
      e^{\hat\beta_1}=e^{0.7}=\mathbf{2.014}
    \]
    --- the odds roughly \emph{double}, by the same factor at every value of $x$.
    \medskip

    \textbf{Why no single number for the probability.} The effect on $p(x)$ depends on where the
    applicant sits on the S-shaped logistic curve: a $0.7$ log-odds shift moves $p$ a lot near
    $p\approx0.5$ but hardly at all when $p$ is close to $0$ or $1$.
  \end{solutionbox}
\end{frame}

\begin{frame}{Problem 7 --- solution (c): why 5- or 10-fold CV}
  \begin{solutionbox}[Solution P7 --- (c) $k$-fold CV vs.\ LOOCV and vs.\ one split]
    \textbf{vs.\ LOOCV.} The $n$ leave-one-out fits are trained on almost identical data sets, so
    their errors are highly positively correlated; averaging strongly correlated quantities reduces
    variance little, so the LOOCV estimate has \emph{higher variance} (and costs $n$ fits rather
    than $k$).
    \medskip

    \textbf{vs.\ a single validation split.} One split is highly variable and \emph{overestimates}
    the test error, because only part of the data trains the model (bias). $k$-fold CV trains on
    $\tfrac{k-1}{k}$ of the data and averages $k$ estimates $\Rightarrow$ less bias and more
    stability.
  \end{solutionbox}
  \begin{takeaway}[Sweet spot]
    \footnotesize $k=5$ or $10$ is the usual compromise: much cheaper and lower-variance than
    LOOCV, yet far less biased and less noisy than a single hold-out split.
  \end{takeaway}
\end{frame}

\begin{frame}{Problem 7 --- solution (d): ridge or lasso?}
  \begin{solutionbox}[Solution P7 --- (d) ridge versus lasso]
    \textbf{Lasso.} The $\ell_1$ penalty can set coefficients \emph{exactly to zero}, so it
    performs \emph{variable selection} --- ideal when only a handful of the $2{,}000$ predictors
    truly matter, and it returns a small, interpretable model.
    \medskip

    \textbf{Why not ridge.} The $\ell_2$ penalty only shrinks coefficients; it keeps all
    $2{,}000$ predictors in the model and tends to lose when most true coefficients are zero.
  \end{solutionbox}
  \begin{takeaway}[Rule of thumb]
    \footnotesize Sparse truth $\to$ lasso (selection); many small effects that all contribute
    $\to$ ridge (shrinkage). The elastic net blends the two.
  \end{takeaway}
\end{frame}

\begin{frame}{Problem 7 --- solution (e): match method to scenario}
  \begin{solutionbox}[Solution P7 --- (e) each method used exactly once]
    \begin{itemize}
      \item[(i)] \textbf{Linear regression} --- quantitative response, goal is \emph{inference}:
        the effect of one extra m$^2$ with a confidence interval, straight from OLS standard errors.
      \item[(ii)] \textbf{Lasso} --- $p=5{,}000\gg n=100$ makes OLS infeasible; the $\ell_1$
        penalty regularises and returns a \emph{short list} of relevant genes.
      \item[(iii)] \textbf{Logistic regression} --- binary response (default yes/no) with
        interpretable coefficients $e^{\hat\beta_j}=$ odds ratios, suitable for the regulator.
      \item[(iv)] \textbf{Random forest} --- captures non-linearities and interactions among
        hundreds of mixed sensors automatically; high accuracy, interpretability not required.
    \end{itemize}
  \end{solutionbox}
\end{frame}
